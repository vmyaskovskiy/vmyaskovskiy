select p.name as служащий, c.name as компания
from person p 
inner join company c on p.company_id = c.id  
where c.id <> 5





select cnt , w.name from ( 
select count(p.id) as cnt , c.name
from      person p 
inner join company c on p.company_id = c.id  
where c.id = p.company_id
group by c.name) w
order by cnt desc
limit 1
